## Quick Start

Write a test file next to your component:

```tsx
// Button.test.tsx
import { describe, it, expect, render, fireEvent, snapshot } from "fieldtest";
import { Button } from "./Button";

describe("Button", () => {
  it("renders the label", async () => {
    const { getByText } = await render(<Button label="Click me" />);
    expect(getByText("Click me")).toBeTruthy();
  });

  it("fires onClick when clicked", async () => {
    let clicked = false;
    const { getByRole } = await render(
      <Button
        label="Click me"
        onClick={() => {
          clicked = true;
        }}
      />,
    );
    await snapshot("before click");
    await fireEvent.click(getByRole("button"));
    expect(clicked).toBe(true);
  });
});
```

Run in the browser:

```bash
fieldtest --ui
```

Or headlessly in Node:

```bash
fieldtest
```
