## expect

Wraps a value and returns a set of matchers to assert its state. Every assertion is recorded in the test result and appears in the Assertions tab of the browser UI.

```ts
import { expect } from "roadtest";

expect(value).toBe(expected);
```

### Type signature

```ts
function expect(received: unknown): Matchers;
```

---

## Equality

### toBe

```ts
toBe(expected: unknown): void
```

Checks strict identity using `Object.is`. Use this for primitives (strings, numbers, booleans). For objects, prefer `toEqual`.

```ts
expect(1 + 1).toBe(2);
expect(user.name).toBe("Alice");
```

### toEqual

```ts
toEqual(expected: unknown): void
```

Deep equality via recursive comparison. Use this for objects and arrays.

```ts
expect({ a: 1 }).toEqual({ a: 1 });
expect([1, 2, 3]).toEqual([1, 2, 3]);
```

On failure, roadtest reports only the paths that differ rather than printing both full objects:

```
toEqual failed:
  - received user.role: "viewer"
  + expected user.role: "admin"
  - received scores[2]: 4
  + expected scores[2]: 3
```

### toStrictEqual

```ts
toStrictEqual(expected: unknown): void
```

Like `toEqual` but also checks that objects have the same type and that `undefined` properties are not ignored. Use this when the constructor or sparse array holes matter.

```ts
expect(new Date("2024-01-01")).toStrictEqual(new Date("2024-01-01"));
expect([1, , 3]).not.toStrictEqual([1, undefined, 3]);
```

## Truthiness

### toBeTruthy

```ts
toBeTruthy(): void
```

Passes for any value that is truthy (`!= null`, `!= 0`, `!= ''`, `!= false`).

```ts
expect("hello").toBeTruthy();
expect(1).toBeTruthy();
```

### toBeFalsy

```ts
toBeFalsy(): void
```

Passes for `0`, `''`, `null`, `undefined`, `false`, and `NaN`.

```ts
expect(0).toBeFalsy();
expect("").toBeFalsy();
```

### toBeNull

```ts
toBeNull(): void
```

Strict `=== null` check.

```ts
expect(result).toBeNull();
```

### toBeUndefined

```ts
toBeUndefined(): void
```

Strict `=== undefined` check.

```ts
expect(user.middleName).toBeUndefined();
```

### toBeDefined

```ts
toBeDefined(): void
```

Passes for any value that is not `undefined`.

```ts
expect(response.data).toBeDefined();
```

## Numeric

### toBeGreaterThan

```ts
toBeGreaterThan(n: number): void
```

```ts
expect(price).toBeGreaterThan(0);
```

### toBeLessThan

```ts
toBeLessThan(n: number): void
```

```ts
expect(responseTime).toBeLessThan(200);
```

### toBeGreaterThanOrEqual

```ts
toBeGreaterThanOrEqual(n: number): void
```

```ts
expect(items.length).toBeGreaterThanOrEqual(1);
```

### toBeLessThanOrEqual

```ts
toBeLessThanOrEqual(n: number): void
```

```ts
expect(discount).toBeLessThanOrEqual(100);
```

### toBeNaN

```ts
toBeNaN(): void
```

Passes when the value is `NaN`.

```ts
expect(Number("not a number")).toBeNaN();
```

### toBeCloseTo

```ts
toBeCloseTo(number: number, numDigits?: number): void
```

Checks that a floating-point number is close to the expected value. `numDigits` (default `2`) controls how many decimal places to round to before comparing.

```ts
expect(0.1 + 0.2).toBeCloseTo(0.3);
expect(Math.PI).toBeCloseTo(3.14159, 5);
```

### toBeTypeOf

```ts
toBeTypeOf(type: string): void
```

Checks the result of `typeof received`.

```ts
expect(42).toBeTypeOf("number");
expect("hello").toBeTypeOf("string");
expect(fetchUser).toBeTypeOf("function");
```

## Strings & Patterns

### toMatch

```ts
toMatch(pattern: string | RegExp): void
```

Checks that the string contains a substring, or matches a regular expression.

```ts
expect(message).toMatch("Welcome");
expect(email).toMatch(/@roadtest\.dev$/);
```

---

## Arrays & Strings

### toContain

```ts
toContain(item: unknown): void
```

Checks that an array includes an item (strict equality), or that a string contains a substring.

```ts
expect(["a", "b", "c"]).toContain("b");
expect("hello world").toContain("world");
```

### toContainEqual

```ts
toContainEqual(item: unknown): void
```

Checks that an array contains an element that deep-equals `item`. Unlike `toContain`, this uses deep equality rather than strict reference equality.

```ts
expect([{ id: 1 }, { id: 2 }]).toContainEqual({ id: 1 });
```

### toHaveLength

```ts
toHaveLength(n: number): void
```

Checks the `.length` property of an array, string, or any value with a `length`.

```ts
expect([1, 2, 3]).toHaveLength(3);
expect("hello").toHaveLength(5);
```

---

## Objects

### toMatchObject

```ts
toMatchObject(expected: object): void
```

Checks that the received object contains all properties in `expected` with matching values. Extra properties are allowed.

```ts
expect(user).toMatchObject({ name: "Alice", role: "admin" });
```

### toBeInstanceOf

```ts
toBeInstanceOf(cls: new (...args: unknown[]) => unknown): void
```

Checks that the value was created with `new cls`.

```ts
expect(error).toBeInstanceOf(Error);
expect(date).toBeInstanceOf(Date);
```

### toHaveProperty

```ts
toHaveProperty(keyPath: string | string[], value?: unknown): void
```

Checks that the object has a property at the given dot-separated key path (or array of keys). Optionally asserts the value at that path.

```ts
expect(user).toHaveProperty("address.city");
expect(user).toHaveProperty("address.city", "Auckland");
expect(user).toHaveProperty(["address", "city"], "Auckland");
```

---

## Functions & Exceptions

### toThrow

```ts
toThrow(message?: string | RegExp | Error): void
```

Checks that a function throws when called. The optional argument asserts the thrown error:

- `string` — error message must contain the string
- `RegExp` — error message must match the pattern
- `Error` instance — must be same constructor and same message

```ts
expect(() => JSON.parse("{")).toThrow();
expect(() => validateAge(-1)).toThrow("must be positive");
expect(() => validateAge(-1)).toThrow(/positive/);
```

## Async matchers

Chain `.resolves` or `.rejects` before a matcher to unwrap a promise first. All sync matchers are available on the async chain.

```ts
await expect(fetchUser("123")).resolves.toEqual({ id: "123", name: "Alice" });
await expect(fetchUser("bad")).rejects.toThrow("not found");
await expect(Promise.resolve(42)).resolves.toBeGreaterThan(0);
```

Negate with `.not` after `.resolves` / `.rejects`:

```ts
await expect(fetchUser("123")).resolves.not.toBeNull();
```

## Spy matchers

These matchers work on spy functions created via `vi.fn()`, `vi.spyOn()`, or returned from a mock factory. See [mock / unmock](/docs/mock) for how to create spies.

### toHaveBeenCalled

```ts
toHaveBeenCalled(): void
```

Passes if the spy was called at least once.

```ts
expect(fetchUser).toHaveBeenCalled();
```

### toHaveBeenCalledTimes

```ts
toHaveBeenCalledTimes(n: number): void
```

Passes if the spy was called exactly `n` times.

```ts
expect(fetchUser).toHaveBeenCalledTimes(2);
```

### toHaveBeenCalledWith

```ts
toHaveBeenCalledWith(...args: unknown[]): void
```

Passes if the spy was called at least once with the given arguments (deep equality).

```ts
expect(fetchUser).toHaveBeenCalledWith("user-123");
```

### toHaveBeenLastCalledWith

```ts
toHaveBeenLastCalledWith(...args: unknown[]): void
```

Passes if the spy's most recent call used the given arguments.

```ts
expect(log).toHaveBeenLastCalledWith("done");
```

### toHaveBeenNthCalledWith

```ts
toHaveBeenNthCalledWith(n: number, ...args: unknown[]): void
```

Passes if the nth call (1-indexed) used the given arguments.

```ts
expect(log).toHaveBeenNthCalledWith(2, "step 2");
```

### toHaveReturnedWith

```ts
toHaveReturnedWith(value: unknown): void
```

Passes if the spy returned the given value in at least one call (deep equality, throws excluded).

```ts
expect(double).toHaveReturnedWith(10);
```

### toHaveReturnedTimes

```ts
toHaveReturnedTimes(n: number): void
```

Passes if the spy returned successfully (without throwing) exactly `n` times.

```ts
expect(double).toHaveReturnedTimes(3);
```

### toHaveLastReturnedWith

```ts
toHaveLastReturnedWith(value: unknown): void
```

Passes if the spy's most recent successful return value equals `value`.

```ts
expect(double).toHaveLastReturnedWith(6);
```

### toHaveNthReturnedWith

```ts
toHaveNthReturnedWith(n: number, value: unknown): void
```

Passes if the nth successful return value (1-indexed, throws excluded) equals `value`.

```ts
expect(double).toHaveNthReturnedWith(2, 4);
```

## Asymmetric matchers

Use inside `toEqual`, `toMatchObject`, or `toHaveBeenCalledWith` to match against a pattern rather than an exact value. All matchers support `.not.toEqual(expect.anything())` etc.

### expect.any

```ts
expect.any(constructor: unknown)
```

Matches any value that is an instance of the given constructor (or the same primitive type for `Number`, `String`, etc.).

```ts
expect(user).toEqual({ id: expect.any(Number), name: expect.any(String) });
```

### expect.anything

```ts
expect.anything();
```

Matches any value except `null` and `undefined`.

```ts
expect(result).toEqual(expect.anything());
```

### expect.arrayContaining

```ts
expect.arrayContaining(arr: unknown[])
```

Matches any array that contains all items in `arr` (extra items are allowed).

```ts
expect(roles).toEqual(expect.arrayContaining(["admin", "editor"]));
```

### expect.objectContaining

```ts
expect.objectContaining(obj: Record<string, unknown>)
```

Matches any object that contains all key-value pairs in `obj` (extra keys are allowed).

```ts
expect(fetchUser).toHaveBeenCalledWith(expect.objectContaining({ id: "123" }));
```

### expect.stringContaining

```ts
expect.stringContaining(str: string)
```

Matches any string that contains `str` as a substring.

```ts
expect(message).toEqual(expect.stringContaining("Welcome"));
```

### expect.stringMatching

```ts
expect.stringMatching(pattern: string | RegExp)
```

Matches any string that contains a substring or matches a regular expression.

```ts
expect(email).toEqual(expect.stringMatching(/@example\.com$/));
```

## Assertion counting

### expect.assertions

```ts
expect.assertions(n: number): void
```

Declares that exactly `n` assertions will run in the test. Fails the test if the count doesn't match — useful for ensuring async code paths are actually reached.

```ts
it("calls the callback once", async () => {
  expect.assertions(1);
  await doWork((result) => {
    expect(result).toBe("ok");
  });
});
```

### expect.hasAssertions

```ts
expect.hasAssertions(): void
```

Declares that at least one assertion will run. Fails the test if no assertions were recorded.

```ts
it("processes each item", async () => {
  expect.hasAssertions();
  for (const item of await loadItems()) {
    expect(item.valid).toBe(true);
  }
});
```

## Custom matchers

### expect.extend

```ts
expect.extend(matchers: Record<string, CustomMatcherFn>): void
```

Adds custom matchers. Each matcher receives the `received` value as its first argument and returns `{ pass: boolean; message: () => string }`.

```ts
expect.extend({
  toBeWithinRange(received: number, min: number, max: number) {
    const pass = received >= min && received <= max;
    return {
      pass,
      message: () => `Expected ${received} to be within [${min}, ${max}]`,
    };
  },
});

expect(50).toBeWithinRange(1, 100);
```

---

## Negation

Prefix any matcher with `.not` to invert the assertion.

```ts
expect(value).not.toBe(expected);
expect(result).not.toBeNull();
expect(list).not.toContain("banned");
expect(fn).not.toHaveBeenCalled();
```

## DOM example

```tsx
const { getByRole } = await render(<Button label="Save" />);
const btn = getByRole("button");

expect(btn.textContent).toBe("Save");
expect(btn.hasAttribute("disabled")).toBeFalsy();
```
